pass-through

new Enumerable([1,2,3,4,5])

where

new Enumerable([1,2,3,4,5])
    .where(function(v) {
        return v < 2;
    })

reverse

new Enumerable([1,2,3,4,5])
    .reverse()

select

new Enumerable([1,2,3,4,5])
    .select(function(v) { 
        return v * 2; 
    })

select-many

new Enumerable([1,2,3,4,5])
    .selectMany(function(v) { 
        return [v, v + 5]; 
    })

select-many-where

new Enumerable([1,2,3,4,5])
    .selectMany(function(v) { 
        return [v, v + 5]; 
    })
    .where(function(v) {
        return v > 5;
    })

projection

new Enumerable([1,2,3,4,5])
    .select(function(v) { 
        return {val: v};
    })

aggregation

new Enumerable([1,2,3,4,5])
    .aggregate(0, function(a, v) {
        return a + v;
    })

to object

new Enumerable([1,2,3,4,5])
    .toDictionary(
        function(o) { return 'key' + o; }
    )
